ZADANIE 1
SELECT NAZWISKO, RPAD(SUBSTR(ETAT, 0, 2), 5, SUBSTR(ID_PRAC, 0, 3)) AS KOD FROM PRACOWNICY;
SELECT NAZWISKO, CONCAT(SUBSTR(ETAT, 0, 2), SUBSTR(ID_PRAC, 0, 3)) AS KOD FROM PRACOWNICY;
SELECT NAZWISKO, SUBSTR(ETAT,1,2)||SUBSTR(ID_PRAC,1,3) AS KOD FROM PRACOWNICY; 

ZADANIE 2
SELECT NAZWISKO, TRANSLATE(NAZWISKO, 'KLM','XXX') AS WOJNA_LITEROM FROM PRACOWNICY;

ZADANIE 3
SELECT NAZWISKO FROM PRACOWNICY WHERE SUBSTR(NAZWISKO,0,TRUNC(LENGTH(NAZWISKO)/2)) LIKE '%L%';

ZADANIE 4
SELECT NAZWISKO, ROUND(PLACA_POD+PLACA_POD*0.15,0) AS PODWYZKA FROM PRACOWNICY; 

ZADANIE 5
SELECT NAZWISKO, PLACA_POD, PLACA_POD*0.2 AS INWESTYCJA, PLACA_POD*0.2*POWER(1.1, 10) AS KAPITAL, PLACA_POD*0.2*POWER(1.1, 10)-PLACA_POD*0.2 AS ZYSK FROM PRACOWNICY; 

ZADANIE 6
SELECT NAZWISKO, TO_CHAR(ZATRUDNIONY, 'YY/MM/DD') AS ZATRUDNI, EXTRACT(YEAR FROM (TO_DATE('01-01-2000', 'DD-MM-YYYY')-zatrudniony) YEAR TO MONTH) AS STAZ_W_2000 FROM PRACOWNICY; 

ZADANIE 7
SELECT NAZWISKO, TO_CHAR(ZATRUDNIONY, 'MONTH, DD YYYY') AS DATA_ZATRUDNIENIA FROM PRACOWNICY WHERE ID_ZESP = 20; 

ZADANIE 8
SELECT TO_CHAR(SYSDATE,'DAY') AS DZIS FROM DUAL; 

ZADANIE 9
SELECT NAZWA, ADRES, CASE 
WHEN ADRES LIKE 'PIOTROWO%' THEN 'NOWE MIASTO'
WHEN ADRES LIKE 'WLODKOWICA%' THEN 'GRUNWALD'
WHEN ADRES LIKE 'MIELZYNSKIEGO%' OR ADRES LIKE 'STRZELECKA%' THEN 'STARE MIASTO' END AS DZIELNICA FROM ZESPOLY;

ZADANIE 10
SELECT NAZWISKO, PLACA_POD, CASE 
WHEN PLACA_POD<480 THEN 'Poniżej '||480 
WHEN PLACA_POD=480 THEN 'Dokładnie '||480 
WHEN PLACA_POD>480 THEN 'Powyżej '||480 
END AS PRÓG FROM PRACOWNICY ORDER BY PLACA_POD DESC;

ZADANIE 11
SELECT NAZWISKO, PLACA_POD, DECODE(SIGN(PLACA_POD-480),  -1 , 'Poniżej '||480, 0 , 'Dokładnie '||480 , 1 , 'Powyżej '||480 ) AS PRÓG FROM PRACOWNICY ORDER BY PLACA_POD DESC;
