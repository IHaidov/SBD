ZADANIE 1
SELECT MIN(PLACA_POD) AS MINIMUM, MAX(PLACA_POD) AS MAKSIMUM, MAX(PLACA_POD)-MIN(PLACA_POD) AS RÓŻNICA FROM PRACOWNICY;

ZADANIE 2
SELECT ETAT, AVG(PLACA_POD) AS SREDNIA FROM PRACOWNICY GROUP BY ETAT ORDER BY SREDNIA DESC ;

ZADANIE 3
SELECT COUNT(ETAT) AS PROFESOROWIE FROM PRACOWNICY WHERE ETAT = 'PROFESOR' GROUP BY ETAT;

ZADANIE 4
SELECT ID_ZESP, SUM(PLACA_POD + NVL(PLACA_DOD,0)) AS SUMARYCZNE_PLACE FROM PRACOWNICY GROUP BY ID_ZESP ORDER BY ID_ZESP ASC;

ZADANIE 5
SELECT MAX(SUM(PLACA_POD + NVL(PLACA_DOD,0))) AS SUMARYCZNE_PLACE FROM PRACOWNICY GROUP BY ID_ZESP;

ZADANIE 6
SELECT ID_SZEFA, MIN(PLACA_POD) AS MINIMALNA FROM PRACOWNICY WHERE ID_SZEFA>0 GROUP BY ID_SZEFA ORDER BY MINIMALNA DESC;

ZADANIE 7
SELECT ID_ZESP, COUNT(ID_PRAC) AS ILU_PRAC FROM PRACOWNICY GROUP BY ID_ZESP ORDER BY ILU_PRAC DESC;

ZADANIE 8
SELECT ID_ZESP, COUNT(ID_PRAC) AS ILU_PRAC FROM PRACOWNICY GROUP BY ID_ZESP HAVING COUNT(ID_PRAC)>3 ORDER BY ILU_PRAC DESC;

ZADANIE 9
SELECT COUNT(*) FROM PRACOWNICY GROUP BY ID_PRAC HAVING COUNT(ID_PRAC)>1;

ZADANIE 10
SELECT ETAT, AVG(PLACA_POD) AS SREDNIA, COUNT(ID_PRAC) AS LICZBA FROM PRACOWNICY WHERE ZATRUDNIONY < TO_DATE('1990-01-01') GROUP BY ETAT ORDER BY ETAT;

ZADANIE 11
SELECT ID_ZESP, ETAT, ROUND(AVG(PLACA_POD+NVL(PLACA_DOD, 0))) AS SREDNIA, ROUND(MAX(PLACA_POD+NVL(PLACA_DOD, 0))) AS MAKSYMALNA FROM PRACOWNICY WHERE ETAT IN ('PROFESOR','ASYSTENT') GROUP BY ID_ZESP, ETAT ORDER BY ID_ZESP ASC;

ZADANIE 12
SELECT DISTINCT EXTRACT(YEAR FROM ZATRUDNIONY) AS ROK, COUNT(NAZWISKO) AS ILU_PRACOWNIKOW FROM PRACOWNICY GROUP BY EXTRACT(YEAR FROM ZATRUDNIONY) ORDER BY ROK;

ZADANIE 13
SELECT LENGTH(NAZWISKO) AS "ILE LITER", COUNT(LENGTH(NAZWISKO)) AS "W ILU NAZWISKACH" FROM PRACOWNICY GROUP BY LENGTH(NAZWISKO) ORDER BY "ILE LITER" ASC; 

ZADANIE 14
SELECT COUNT(NAZWISKO) AS "Ile nazwisk z A" FROM PRACOWNICY WHERE NAZWISKO LIKE '%A%' OR NAZWISKO LIKE '%a%';

ZADANIE 15
SELECT DISTINCT (SELECT COUNT(NAZWISKO) AS "Ile nazwisk z A" FROM PRACOWNICY WHERE NAZWISKO LIKE '%A%' OR NAZWISKO LIKE '%a%') AS "ILE NAZWISK Z A", (SELECT COUNT(NAZWISKO) AS "Ile nazwisk z E" FROM PRACOWNICY WHERE NAZWISKO LIKE '%e%' OR NAZWISKO LIKE '%E%') AS "ILE NAZWISK Z E"FROM PRACOWNICY;  

ZADANIE 16
SELECT ID_ZESP, SUM(PLACA_POD) AS SUMA_PLAC, LISTAGG(NAZWISKO||':'||PLACA_POD,',') WITHIN GROUP(ORDER BY NAZWISKO) AS PRACOWNICY FROM PRACOWNICY GROUP BY ID_ZESP ORDER BY ID_ZESP;
